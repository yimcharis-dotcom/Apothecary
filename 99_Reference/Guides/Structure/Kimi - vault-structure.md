## Quick Setup Script

Save this as `setup-vault.sh` (Mac/Linux) or `setup-vault.bat` (Windows):

### setup-vault.sh
```bash
#!/bin/bash

VAULT_NAME="AI-Dev-Wiki"

# Create main directories
mkdir -p "$VAULT_NAME"/{0-Inbox,1-Projects,2-Prompts,3-Knowledge,4-Reviews,5-References,6-Templates,7-Archive}

# Create subdirectories
cd "$VAULT_NAME"

# Project subdirectories
for project in system-spec-generator grammar-checker custom-gpt-projects; do
    mkdir -p "1-Projects/$project"/{prompts,examples,reviews,resources}
done

# Prompt subdirectories
mkdir -p 2-Prompts/{by-category/{code-generation,technical-writing,analysis,creative,problem-solving},by-model/{gpt-4,claude,gemini,custom-gpt},templates/{system-prompts,few-shot-examples,prompt-chains},0-Inbox}

# Knowledge subdirectories  
mkdir -p 3-Knowledge/{ai-concepts,tools-and-frameworks,best-practices}

# Review subdirectories
mkdir -p 4-Reviews/{ai-output-reviews,project-retrospectives,improvement-logs}

# Reference subdirectories
mkdir -p 5-References/{bookmarks,articles,code-snippets}

# Template subdirectories
mkdir -p 6-Templates/templater-scripts

# Archive subdirectories
mkdir -p 7-Archive/{2024,2025}

# Create index files
touch 0-Inbox/{dashboard,daily-capture,ideas}.md
touch 1-Projects/index.md
touch 2-Prompts/index.md
touch 3-Knowledge/index.md
touch 4-Reviews/index.md
touch 5-References/index.md
touch 6-Templates/index.md
touch 7-Archive/index.md

echo "Vault structure created successfully!"
echo "Now copy the template files into their respective directories."
```

### setup-vault.bat
```batch
@echo off
set VAULT_NAME=AI-Dev-Wiki

REM Create main directories
mkdir "%VAULT_NAME%\0-Inbox"
mkdir "%VAULT_NAME%\1-Projects"
mkdir "%VAULT_NAME%\2-Prompts"
mkdir "%VAULT_NAME%\3-Knowledge"
mkdir "%VAULT_NAME%\4-Reviews"
mkdir "%VAULT_NAME%\5-References"
mkdir "%VAULT_NAME%\6-Templates"
mkdir "%VAULT_NAME%\7-Archive"

cd "%VAULT_NAME%"

REM Create project subdirectories
for %%p in (system-spec-generator,grammar-checker,custom-gpt-projects) do (
    mkdir "1-Projects\%%p\prompts"
    mkdir "1-Projects\%%p\examples"
    mkdir "1-Projects\%%p\reviews"
    mkdir "1-Projects\%%p\resources"
)

REM Create prompt subdirectories
mkdir "2-Prompts\by-category\code-generation"
mkdir "2-Prompts\by-category\technical-writing"
mkdir "2-Prompts\by-category\analysis"
mkdir "2-Prompts\by-category\creative"
mkdir "2-Prompts\by-category\problem-solving"
mkdir "2-Prompts\by-model\gpt-4"
mkdir "2-Prompts\by-model\claude"
mkdir "2-Prompts\by-model\gemini"
mkdir "2-Prompts\by-model\custom-gpt"
mkdir "2-Prompts\templates\system-prompts"
mkdir "2-Prompts\templates\few-shot-examples"
mkdir "2-Prompts\templates\prompt-chains"
mkdir "2-Prompts\0-Inbox"

REM Create other subdirectories
mkdir "3-Knowledge\ai-concepts"
mkdir "3-Knowledge\tools-and-frameworks"
mkdir "3-Knowledge\best-practices"

mkdir "4-Reviews\ai-output-reviews"
mkdir "4-Reviews\project-retrospectives"
mkdir "4-Reviews\improvement-logs"

mkdir "5-References\bookmarks"
mkdir "5-References\articles"
mkdir "5-References\code-snippets"

mkdir "6-Templates\templater-scripts"

mkdir "7-Archive\2024"
mkdir "7-Archive\2025"

REM Create index files
echo. > 0-Inbox\dashboard.md
echo. > 0-Inbox\daily-capture.md
echo. > 0-Inbox\ideas.md
echo. > 1-Projects\index.md
echo. > 2-Prompts\index.md
echo. > 3-Knowledge\index.md
echo. > 4-Reviews\index.md
echo. > 5-References\index.md
echo. > 6-Templates\index.md
echo. > 7-Archive\index.md

echo Vault structure created successfully!
echo Now copy the template files into their respective directories.
```

---

**Usage:**
1. Run the setup script for your OS
2. Copy all template files from `/templates/` into `6-Templates/`
3. Copy the dashboard content into `0-Inbox/dashboard.md`
4. Start Obsidian and open your vault
5. Begin with the Quick Start Guide!
